#include <iostream>

template<class T1,class T2,class T3>
class combine{
private:
	T1 a;
	T2 b;
public:
	combine(T1 x,T2 y):a(x),b(y) { }
	T3 operator()(T3 x){
		return a(a(x)+b(x));
	}
};
int main()
{

    return 0;
