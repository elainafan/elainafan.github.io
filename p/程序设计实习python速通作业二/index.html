def deepcopy(a):
    if isinstance(a,list):
        tem=[]
        for item in a:
            tem.append(deepcopy(item))
        return tem
    elif isinstance(a,tuple):
        tem=tuple(deepcopy(item) for item in a)
        return tem
    else:
        return a
