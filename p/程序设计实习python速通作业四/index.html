def mysum(x):
    def inner(y=None):
        if y is None:
            return x
        return mysum(x + y)
    return inner

#pylint: disable = no-value-for-parameter
def accfunc(f):
    def inner(g=None):
        if g is None:
            return f
        else:
            return accfunc(lambda x: g(f(x)))
    return inner
def f1(x):
    return x + 1
def f2(x):
    return x * x
def f3(x):
    return x + x
def f4(x):
    return x*3
def f5(x):
    return x-4

while True:
    try:
        s = input()
        n = int(input())
        s = s.split()
        k = accfunc
        for x in s:
            k = k(eval(x))
        print(k()(n))
    except:  #读到 eof产生异常
        break

exit = None
def times(n):
    num = 0
    while True:
        yield num
        num += n
n,m = map(int,input().split())
seq = times(n)
if str(type(seq) == "<class 'generator'>"):
	i = 0
	for x in seq:
		print(x)
		i += 1
		if i == m:
			break
